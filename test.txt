    def selectMat(self,matrix,startPoint, endPoint):
        new_mat = [[None]*(endPoint[1]-startPoint[1])]*(endPoint[0]-startPoint[0])
        for x in range((endPoint[0]-startPoint[0])):
            for y in range((endPoint[1]-startPoint[1])):
                new_mat[x][y] = matrix[startPoint[0]+x][startPoint[1]+y]
        return new_mat

    def selectSprite(self,sheetNum,sheetList,x,y,dispX,dispY) :
        self.img = Image.open(self.rootPath+sheetList[sheetNum][0]).convert("RGBA")
        step_x = (sheetList[sheetNum][1]/sheetList[sheetNum][3])
        step_y = (sheetList[sheetNum][2]/sheetList[sheetNum][4])
        box = ([int(math.floor(x*step_x)b  ),int(math.floor(y*step_y)),int(math.floor((x+1)*step_x)),int(math.floor((y+1)*step_y))])
        #print([x*step_x,y*step_y,(x+1)*step_x,(y+1)*step_y])
        self.img = self.img.crop(box)
        self.img2 = ImageTk.PhotoImage(self.img)
        self.spriteListMem.append(self.img2)
        self.canvas.create_image(dispX,dispY, image=self.img2, anchor="nw")
		
		    def drawBackgroundSheet(self,backSettings,posX,posY) :
        img = backSettings[0]
        img2 = ImageTk.PhotoImage(img.resize((int(backSettings[1]*self.zoom),int(backSettings[2]*self.zoom)), resample=self.resampleType))
        self.spriteListMem.append(img2)
        self.canvas.create_image(posX,posY, image=img2, anchor="nw")
		

    def drawSheet(self,sheet,grid=True) :
        self.img5 = ImageTk.PhotoImage(sheet[0])
        self.canvas.create_image(0,0, image=self.img5, anchor="nw")
        if grid :
            for x in range(sheet[3]+1) :
                new_x = int(x*(sheet[1]/sheet[3]))
                new_x2 = int((x+0.5)*(sheet[1]/sheet[3]))
                self.canvas.create_line(new_x, 0, new_x, self.height, fill="black")    
                self.canvas.create_line(new_x2, 0, new_x2, self.height, fill="gray")    
            for y in range(sheet[4]+1) :
                new_y = int(y*(sheet[2]/sheet[4]))
                new_y2 = int((y+0.5)*(sheet[2]/sheet[4]))
                self.canvas.create_line(0, new_y, self.width, new_y, fill="black")     
                self.canvas.create_line(0, new_y2, self.width, new_y2, fill="gray")
        self.canvas.pack()
        self.root.update()
        self.root.mainloop()
